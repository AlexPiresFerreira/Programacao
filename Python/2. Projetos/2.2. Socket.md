```
import socket
import sys

def main():
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    except socket.error as e:
        print("A conexão falhou!!!")
        print(f"Erro: {e}")
        sys.exit()

    print("Socket criado com sucesso")

    HostAlvo = input("Digite o Host ou IP a ser conectado: ")
    PortaAlvo = input("Digite a porta a ser conectada: ")

    try:
        s.connect((HostAlvo, int(PortaAlvo)))
        print(f"Cliente TCP conectado com sucesso no Host: {HostAlvo} na Porta: {PortaAlvo}")

        # Enviando mensagem para o servidor
        mensagem = input("\nDigite a mensagem para enviar ao servidor: ")
        s.send(mensagem.encode())
        print(f"Cliente enviou: {mensagem}")

        # Recebendo resposta do servidor
        resposta = s.recv(4096)
        print(f"Servidor respondeu: {resposta.decode()}")

    except socket.error as e:
        print(f"Não foi possível conectar no Host: {HostAlvo} na Porta: {PortaAlvo}")
        print(f"Erro: {e}")
        sys.exit()
    finally:
        print("\nCliente: Fechando conexão...")
        s.close()

if __name__ == "__main__":
    main()

```

- ServerTCP

```
import socket
import sys

def main():
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    except socket.error as e:
        print("Erro ao criar socket!")
        print(f"Erro: {e}")
        sys.exit()

    print("Socket do servidor criado com sucesso")

    host = 'localhost'  # ou '0.0.0.0' para aceitar de qualquer IP
    porta = 5433

    try:
        s.bind((host, porta))
    except socket.error as e:
        print(f"Erro no bind na porta {porta}")
        print(f"Erro: {e}")
        sys.exit()

    s.listen(5)  # Aceita até 5 conexões na fila
    print(f"Servidor TCP escutando em {host}:{porta}")
    print("Aguardando conexões... (Ctrl+C para encerrar)")

    try:
        while True:
            conexao, endereco = s.accept()
            print(f"\nConexão estabelecida com: {endereco[0]}:{endereco[1]}")

            try:
                # Recebendo dados do cliente
                dados = conexao.recv(4096)
                if dados:
                    mensagem_recebida = dados.decode()
                    print(f"Cliente enviou: {mensagem_recebida}")

                    # Enviando resposta ao cliente
                    resposta = f"Servidor recebeu sua mensagem: '{mensagem_recebida}'"
                    conexao.send(resposta.encode())
                    print("Resposta enviada ao cliente")

            except Exception as e:
                print(f"Erro na comunicação: {e}")
            finally:
                conexao.close()
                print(f"Conexão com {endereco[0]} encerrada")

    except KeyboardInterrupt:
        print("\n\nServidor: Encerrando...")
    finally:
        s.close()
        print("Servidor: Socket fechado com sucesso!")

if __name__ == "__main__":
    main()

```

import socket

s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
s.settimeout(1.0)  # timeout de 1 segundo
print("Socket Criado com sucesso")

host = "localhost"
port = 5432
s.bind((host, port))

mensagem = "\nServidor: Olá Cliente e ai beleza"
print(f"Servidor escutando na porta {port}...")
print("Pressione Ctrl+C para encerrar")

try:
    while True: 
        try:
            dados, end = s.recvfrom(4096)
            if dados: 
                print(f"Servidor recebeu: {dados.decode()}")
                print("Servidor enviando mensagem...")
                s.sendto(dados + (mensagem.encode()), end)
        except socket.timeout:
            # Timeout acontece a cada 1 segundo, permitindo checar o Ctrl+C
            continue
except KeyboardInterrupt:
    print("\n\nServidor: Encerrando conexão...")
finally:
    s.close()
    print("Servidor: Socket fechado com sucesso!")

### Python — testes simples de rede com `socket`

```python
import socket
import os

sites = open('socket.txt', 'r')

for ende in sites.readlines():
    host = ende.strip()
    print("IP:", socket.gethostbyname(host))
    print("HOSTNAME, ALIASLIST, IPADDRLIST:", socket.gethostbyname_ex(host))
    os.system(f"ping {host}")
```

**Ideias importantes:**
- `socket.gethostbyname(host)` → resolve hostname para IP.
- `socket.gethostbyname_ex(host)` → retorna (hostname, aliaslist, ipaddrlist).
- Útil para **verificar DNS e conectividade** de vários hosts em sequência.
