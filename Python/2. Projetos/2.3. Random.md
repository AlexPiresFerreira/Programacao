
- Gerador de senha 

```
import random 
import string

tamanho = int(input("Digite o tamanho de senha que você deseja: "))

chars = string.ascii_letters + string.digits + '!@#$%&*()-+_,.'

rnd = random.SystemRandom()

print(''.join(rnd.choice(chars) for i in range(tamanho)))
```

- Comparador de hash

```
import hashlib

# 2 arquivos para comparar 

arquivo1 = [novo_do_arquivo1]
arquivo2 = [novo_do_arquivo2]

hash1 = hashlib.new('hash(md5, sha1 e etc)')
hash1.update(open(arquivo1, 'rb').read())

hash2 = hashlib.new('hash(md5, sha1 e etc)')
hash2.update(open(arquivo2, 'rb').read())

if hash1.digest() != hash2.digest(): 
	print(f"O arquivo: {arquivo1} é diferente do arquivo {arquivo2}")
	print("O hash do arquivo a.txt é: ", hash1.hexdigest())
	print("O hash do arquivo b.txt é: ", hash2.hexdigest())
else: 
	print(f"O arquivo: {arquivo1} é igual do arquivo {arquivo2}")
	print("O hash do arquivo a.txt é: ", hash1.hexdigest())
	print("O hash do arquivo b.txt é: ", hash2.hexdigest())
```

- Gerador de hash 

```
import hashlib

userHash = input("Digite o texto a ser gerado o hash: ")

menu = int(input(''' ### MENU - ESCOLHA O TIPO DE HASH ###
				1) MD5
				2) SHA1
				3) SHA256
				4) SHA512
				Digite o número do hash a ser gerado: '''))

if menu == 1:
	resultado = hashlib.md5(userHash.encode('utf-8'))
	print(f'O hash MD5 da string "{userHash}" é: ', resultado.hexdigest())
elif menu == 2:
	resultado = hashlib.sha1(userHash.encode('utf-8'))
	print(f'O hash sha1 da string "{userHash}" é: ', resultado.hexdigest())	 
elif menu == 3:
	resultado = hashlib.sha256(userHash.encode('utf-8'))
	print(f'O hash sha256 da string "{userHash}" é: ', resultado.hexdigest())	
elif menu == 4:
	resultado = hashlib.sha512(userHash.encode('utf-8'))
	print(f'O hash sha512 da string "{userHash}" é: ', resultado.hexdigest())	
else:
	print("Opção errada, ecolha uma opção valida.")
```
